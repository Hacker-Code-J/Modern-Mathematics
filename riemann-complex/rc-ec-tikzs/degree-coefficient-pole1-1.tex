\node[draw=black, align=left] at (5.75,-5.5) {

	$\displaystyle\xRightarrow[{t\in[0,2\pi]}]{z=e^{it}}-\oint_{[0,2\pi]}\left(e^{-it}\right)\left(ie^{it}\; \mathrm{d}t\right)=-i\oint_{[0,2\pi]}1\; dt=2\pi i\cdot (-1)$
%	\\ [10pt]
