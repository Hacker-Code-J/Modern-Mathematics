\documentclass[tikz,border=3.14mm]{standalone}
\usepackage{pgfplots}
\pgfplotsset{compat=1.16,width=16cm}
\tikzset{
	declare function={
		fluctuating_r(\u,\v,\r)=\r + 0.6*sin(3*\u)*cos(4*\v); % Increased amplitude and frequency
		torusx(\u,\v,\R,\r)=cos(\u)*(\R + fluctuating_r(\u,\v,\r)*cos(3*\v));
		torusy(\u,\v,\R,\r)=(\R + fluctuating_r(\u,\v,\r)*cos(2*\v))*sin(\u);
		torusz(\u,\v,\R,\r)=fluctuating_r(\u,\v,\r)*sin(7*\v);
},
declare function={
	fluctuating_r2(\u,\v,\r)=\r + 0.7*sin(2*\u)*cos(3*\v); % Increased amplitude and frequency
	torusx2(\u,\v,\R,\r)=cos(\u)*(\R + fluctuating_r2(\u,\v,\r)*cos(2*\v));
	torusy2(\u,\v,\R,\r)=(\R + fluctuating_r2(\u,\v,\r)*cos(3*\v))*sin(\u);
	torusz2(\u,\v,\R,\r)=fluctuating_r2(\u,\v,\r)*sin(5*\v);
}}
\begin{document}
	\begin{tikzpicture}
		\pgfmathsetmacro{\R}{4} % Major radius
		\pgfmathsetmacro{\r}{1.2} % Minor radius
		\pgfplotsset{view={35}{60},axis lines=none,}
		\begin{axis}[colormap/cool]
			% Generate the fluctuating torus mesh
			\pgfplotsinvokeforeach{0,10,...,350} {
				\addplot3[
				samples y=0,
				domain=0:361,
				smooth,
				samples=100,
				thin,
				surf,
				z buffer=sort
				%				mesh
				]  
				(
				{torusx(x,#1+x/12,\R,\r)}, 
				{torusy(x,#1+x/12,\R,\r)}, 
				{torusz(x,#1+x/12,\R,\r)}
				);
				\addplot3[
				samples y=0,
				domain=0:360,
				smooth,
				samples=100,
				thin,
				surf,
				z buffer=sort
				%				mesh
				]  
				(
				{torusx(#1,x,\R,\r)}, 
				{torusy(#1,x,\R,\r)}, 
				{torusz(#1,x,\R,\r)}
				);
			}
		\end{axis}
		\begin{scope}[shift={(-4.1,-4.1)}]
		\pgfmathsetmacro{\R}{4} % Major radius
		\pgfmathsetmacro{\r}{1.2} % Minor radius
		\pgfplotsset{view={35}{60},axis lines=none,}
		\begin{axis}[colormap/cool]
			% Generate the fluctuating torus mesh
			\pgfplotsinvokeforeach{0,10,...,350} {
				\addplot3[
				samples y=0,
				domain=0:361,
				smooth,
				samples=100,
				thin,
				surf,
				z buffer=sort
				%				mesh
				]  
				(
				{torusx2(x,#1+x/12,\R,\r)}, 
				{torusy2(x,#1+x/12,\R,\r)}, 
				{torusz2(x,#1+x/12,\R,\r)}
				);
				\addplot3[
				samples y=0,
				domain=0:360,
				smooth,
				samples=100,
				thin,
				surf,
				z buffer=sort
				%				mesh
				]  
				(
				{torusx2(#1,x,\R,\r)}, 
				{torusy2(#1,x,\R,\r)}, 
				{torusz2(#1,x,\R,\r)}
				);
			}
		\end{axis}
		\end{scope}
	\end{tikzpicture}
\end{document}
