\begin{tikzpicture}
	% Define colors for aesthetic appeal
	\definecolor{upperboundcolor}{RGB}{0, 153, 255}   % Light blue for upper bound
	\definecolor{lowerboundcolor}{RGB}{255, 102, 102}  % Soft red for lower bound
	\definecolor{setcolor}{RGB}{0, 204, 102}           % Bright green for elements in set S
	\definecolor{boundarc}{RGB}{153, 153, 153}         % Gray for the arc of bounds
	
	% Define center and radius for radial layout
	\coordinate (center) at (0,0);
	\def\radius{4}
	
	% Draw the arc to represent the number line bounds
	\draw[thick, color=boundarc] 
	(center) ++(180:\radius) arc[start angle=180, end angle=0, radius=\radius];
	
	% Mark lower bound with a larger dot and label
	\node[fill=lowerboundcolor, circle, minimum size=5pt, label=below:{\textbf{Lower Bound}}] 
	(LowerBound) at ($(center)+(180:\radius)$) {};
	
	% Mark upper bound with a larger dot and label
	\node[fill=upperboundcolor, circle, minimum size=5pt, label=below:{\textbf{Upper Bound}}] 
	(UpperBound) at ($(center)+(0:\radius)$) {};
	
	% Add set S elements as gradient dots along the arc between bounds
	\foreach \angle [count=\i] in {150, 135, 120, 105, 90, 75, 60, 45, 30} {
		% Use different shades of green for each point
		\node[fill=setcolor!\i0!black, circle, minimum size=3pt] at ($(center)+(\angle:\radius)$) {};
	}
	
	% Labels for the significance of bounds
	\node[align=center, below=3em] at ($(center)+(180:\radius) - (0, 1.2)$) 
	{No elements \\ left of \\ lower bound};
	\node[align=center, below=3em] at ($(center)+(0:\radius) - (0, 1.2)$) 
	{No elements \\ beyond \\ upper bound};
	
	% Brace decorations for the arc with descriptions
	\draw[decorate, decoration={brace, amplitude=10pt, mirror}, thick, color=lowerboundcolor] 
	($(center) + (170:\radius + 0.2)$) arc[start angle=170, end angle=150, radius=\radius + 0.2]
	node[midway, left=0.7em, color=lowerboundcolor] {\footnotesize Lower Bound Region};
	
	\draw[decorate, decoration={brace, amplitude=10pt}, thick, color=upperboundcolor] 
	($(center) + (30:\radius + 0.2)$) arc[start angle=30, end angle=10, radius=\radius + 0.2]
	node[midway, right=0.7em, color=upperboundcolor] {\footnotesize Upper Bound Region};
	
	% Central label for set S
	\node[align=center, color=setcolor, font=\bfseries] at (center) 
	{\Large Set \( S \) \\ \footnotesize All points \\ between bounds};
	
\end{tikzpicture}
