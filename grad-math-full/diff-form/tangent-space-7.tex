			\pgfmathparse{\ycenter + \i*\ystep}

			(axis cs:\xx, {\ycenter - \ycount*\ystep/2})
			-- (axis cs:\xx, {\ycenter + \ycount*\ystep/2});
		}

	\draw[-Stealth] (\a-2.5,\ay) to (\a+2.5,\ay) node[right] {$\mathrm{d}x$};
