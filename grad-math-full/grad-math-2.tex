\defbox[Relation]{\begin{definition*}
	$\mathcal{R}\subseteq A\times B$
\end{definition*}}

\begin{example*}
	Let $A=\set{1,2}$ and $B=\set{4,5}$. Then \[
	A\times B=\set{(1,4),(1,5),(2,4),(2,5)}.
	\] Here, $R=\set{(1,4),(2,5)}\subseteq A\times B$ be a relation.
\end{example*}

\defbox[Equivalence Relation]{\begin{definition*}
	Let $R$ be a relation on $A$.
	\begin{enumerate}[(i)]
		\item (Reflexivity)
		\item (Symmetry)
		\item (Transitivity)
	\end{enumerate}
\end{definition*}}

\probox{\begin{proposition}
	Let $X$ and $Y$ are sets. Let $f:X\to Y$ be \begin{enumerate}[(1)]
		\item If $g\circ f$ is one-to-one, then $f$ is one-to-one.
		\item If $g\circ f$ is onto, then $g$ is onto.
	\end{enumerate}
\end{proposition}}

\probox{\begin{proposition}
	Let $X$ and $Y$ are sets. Let $f:X\to Y$ be \begin{enumerate}[(1)]
		\item If $f$ and $g$ are both one-to-one, then $g\circ f$ is one-to-one.
		\item If $f$ and $g$ are both one-to-one, then $g\circ f$ is onto.
	\end{enumerate}
\end{proposition}}
